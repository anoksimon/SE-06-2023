#!/bin/bash
# Write A shell script that does the following:

# 1: Asks for the user name, cohort, score,
# 2: If the score is less than 80, print out "You're Out"
# 3: If score is greater than 80, print out "You're In"

score="80"
echo "What is your name"
read name
echo "Which Cohort are you"
read Cohort
echo "score"
read score
if ["$score" -lt 80 || "$score" -ge 80];
then
    echo "$name in Cohort$Cohort You're Out."
else
    echo "$name in Cohort$Cohort You're In."
fi
